27      public UserInterface() {
28          this.game = new Game();
29          this.player = new Player();
30          this.ai = new Ai();

32      }

34      public void greet() {
35          System.out.println("Welcome to Texas Fold'em!");
36          go();
37      }

39      public void go() {
40          setGameSpeed(askGameSpeed());
41          game.preparePack();
42          game.setPlayerChips();
43          game.setAiChips();
44          newRound();
45      }
46  
47      public void bettingInstructions() {
48          System.out.println("type 'c' for call/check, 'r' for raise 'f' for fold");
49      }

51      public void chipSituation() {
52          System.out.println("There is " + game.getPotSize() + " chips in the pot");
53          System.out.println("AI has " + game.aiChipsLeft() + "chips left");
54          System.out.println("You have " + game.playerChipsLeft() + "chips left");
55      }

65      public void newRound() {
66          prepareForNewRound();
67          shuffle();
68          blinds();
69          pocketCards();
70          chipSituation();
71          bettingRound();
72          flop();
73          chipSituation();
74          bettingRound();
75          turn();
76          chipSituation();
77          bettingRound();
78          river();
79          chipSituation();
80          bettingRound();
81      }

98      public void playerRaise() {
99          System.out.println("How much?");
100         while (true) {
101             double amount = Integer.parseInt(scanner.use());
102             if (amount >= 2
103                     * game.getBettingHistory().get(game.getBettingHistory().size()
104                             - 1) && amount >= game.getBigBlind()) {
105                 System.out.println(game.raise(amount));
106                 break;
107             } else {
108                 System.out.println("You must raise atleast twice "
109                         + "the amount of last bet");
110             }
111         }
112     }
113 
114     public void playerAction() {
115         String action = scanner.use();
116         if (action.equals("c")) {
117             System.out.println(game.checkOrCall());
118         } else if (action.equals("r")) {
119             playerRaise();
120         } else if (action.equals("f")) {
121             aiWinsRound();
122             newRound();
123         } else {
124             playerAction();
125         }
126     }

140     public void aiBets(String action) {

144     public void aiAllIn(String action) {

147 
148     public void aiAction() {
149         String action = game.aiAction();
150         if (action.equals("AI folds")) {
151             playerWinsRound();
152             System.out.println(action);
153             newRound();
154         } else if (action.equals("AI calls")) {
155             aiCalls();
156         } else if (action.contains("bet")) {
157             aiBets(action);
158             playerAction();
159         } else if (action.contains("all-in")) {
160             aiAllIn(action);
161         }
162     }
163 
164     public void bettingRound() {
165         if (!game.bettingOrder()) {
166             bettingInstructions();
167             playerAction();
168             aiAction();
169         } else {
170             aiAction();
171             bettingInstructions();
172             playerAction();
173         }
174     }
175 
176     public String askGameSpeed() {
177         System.out.println("Type 'fast', 'normal' or 'slow' to determine game speed");
178         String answer = scanner.use();
179         return answer;
180     }
181 
182     public void setGameSpeed(String answer) {
183         if (answer.contains("fast")) {
184             game.setBigBlind(30);
185             game.setStackSize(500);
186         } else if (answer.contains("normal")) {
187             game.setBigBlind(30);
188             game.setStackSize(1000);
189         } else if (answer.contains("slow")) {
190             game.setBigBlind(30);
191             game.setStackSize(2000);
192         } else {
193             System.out.println("I didn't understand. Let's try again.");
194             go();
195         }
196     }

209     }
