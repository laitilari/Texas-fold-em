6   package tfe.core.game;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  import tfe.core.ai.Ai;
11  import tfe.core.cards.Card;
12  import tfe.core.cards.PackOfCards;
13  import tfe.core.cards.TableCards;
14  import tfe.core.player.Player;
15  
16  /**
17   *
18   * @author ilarilai
19   */
20  public class Dealer {
21  
22      private PackOfCards pack;
23      private String dealerName;
24  
25      public Dealer() {
26          this.dealerName = "James";
27          this.pack = new PackOfCards();
28      }
29  
30      public void assemblePack() {
31          pack.assemblePack();
32      }
33  
34      public void shufflePack() {
35          pack.shuffle();
36      }
37  
38      public void reAssemblePack() {
39          pack.reAssemblePack();
40      }
41  
42      public void dealPocketCards(Player player, Ai ai) {
43          List<Card> playerCards = pack.takeMany(2);
44          List<Card> aiCards = pack.takeMany(2);
45          player.drawPocketCards(playerCards);
46          ai.drawPocketCards(aiCards);
47      }
48  
49      public void dealFlop(TableCards table) {
50          List<Card> flop = pack.takeMany(3);
51          table.drawFlop(flop);
52      }
53  
54      public void dealTurn(TableCards table) {
55          table.drawCard(dealCard());
56      }
57  
58      public void dealRiver(TableCards table) {
59          table.drawCard(dealCard());
60      }
61  
62      public Card dealCard() {
63          Card card = pack.takeOne();
64          return card;
65      }
66  
67      public String getDealerName() {
68          return this.dealerName;
69      }
70  }
